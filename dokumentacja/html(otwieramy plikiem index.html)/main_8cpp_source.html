40 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);

42 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
43
44 // Specify texture image
45 if (nrChannels == 3) {
46 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGB, GL_UNSIGNED_BYTE, data);
47 }
48 else if (nrChannels == 4) {
49 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, data);
50 }
51 glGenerateMipmap(GL_TEXTURE_2D);
52 }
53 else {
54 std::cerr << "Failed to load texture: " << stbi_failure_reason() << '\n';
55 }
56
57 stbi_image_free(data);

615void renderScene(const Shape& shape1, const Shape& shape2, const Shape& shape3) {
616 glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
617 glLoadIdentity();
618
619 // Set up light properties
620 GLfloat lightPosition[] = { 0.0f, 0.0f, 5.0f, 1.0f }; // Light position in eye space
621 GLfloat lightAmbient[] = { 0.2f, 0.2f, 0.2f, 1.0f }; // Ambient light color
622 GLfloat lightDiffuse[] = { 1.0f, 1.0f, 1.0f, 1.0f }; // Diffuse light color
623 GLfloat lightSpecular[] = { 1.0f, 1.0f, 1.0f, 1.0f }; // Specular light color
624
625 glEnable(GL_LIGHTING);
626 glEnable(GL_LIGHT0);
627
628
629 glLightfv(GL_LIGHT0, GL_POSITION, lightPosition);
630 glLightfv(GL_LIGHT0, GL_AMBIENT, lightAmbient);
631 glLightfv(GL_LIGHT0, GL_DIFFUSE, lightDiffuse);
632 glLightfv(GL_LIGHT0, GL_SPECULAR, lightSpecular);
633
634 // Enable material properties
635 glEnable(GL_COLOR_MATERIAL);
636 glColorMaterial(GL_FRONT, GL_AMBIENT_AND_DIFFUSE);
637 glEnable(GL_TEXTURE_2D);
638 glEnable(GL_BLEND);
639 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
640
641 // Render cube
642 glm::mat4 projection = glm::perspective(glm::radians(30.0f), 1.0f, 0.1f, 100.0f);
643 glMatrixMode(GL_PROJECTION);
644 glLoadMatrixf(glm::value_ptr(projection));
645
646 glm::mat4 view = glm::lookAt(glm::vec3(cameraX - 2.0f, cameraY, cameraZ + 5.0f), glm::vec3(0.0f, 0.0f, 3.0f), glm::vec3(0.0f, 1.0f, 0.0f));
647 GLfloat* viewMatrixPtr = glm::value_ptr(view);
648 glMatrixMode(GL_MODELVIEW);
649 glLoadMatrixf(viewMatrixPtr);
650
651 shape1.draw();
652
653 // Render pyramid
654 view = glm::lookAt(glm::vec3(cameraX + 1.0f, cameraY, cameraZ + 4.0f), glm::vec3(0.0f, 0.0f, 0.0f), glm::vec3(0.0f, 1.0f, 0.0f));
655 viewMatrixPtr = glm::value_ptr(view);
656 glMatrixMode(GL_MODELVIEW);
657 glLoadMatrixf(viewMatrixPtr);
658
659 shape2.draw();
